    ForEach-Object {
        $content = Get-Content $_.FullName -Raw
        $plainPhones = ([regex]::Matches($content, '(?<!href="tel:)\(437\)\s*747-6737|(?<!href="tel:)437-747-6737(?!")') | Measure-Object).Count

            Write-Host "FAIL: $($_.Name) has $plainPhones unlinked phones"
        }
    }
